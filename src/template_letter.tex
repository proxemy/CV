\documentclass[%
	fontsize=12,
	DIV=15,
	BCOR=0pt,
	fromalign=right,
	version=last,
	parskip=half
]{scrlttr2}
% BCOR: sets binding correction margin for bound paper
% DIV: the number of page divisions (1 head, 2 foot, rest body).

\usepackage[
	colorlinks=true,
	urlcolor=blue,
	linkcolor=black,
	unicode=true
]{hyperref}

\LoadLetterOption{DIN}
\setkomavar{fromname}{\MyNameShort}
\setkomavar{fromaddress}{%
	\MyStreet{}\\
	\MyCityCode{} \MyCity\\
	\medskip
	% adds a table with remote contact info below the general senders address
	\begin{tabular}{@{}r|r@{}}
		\MyContact
	\end{tabular}
}

% set senders address. a small line ontop of recipient address in the letter window
\setkomavar{backaddressseparator}{\enspace\textperiodcentered\enspace}
\setkomavar{backaddress}{\MyNameFirst{} \MyNameLast{}\\ \MyStreet{}\\ \MyCityCode{} \MyCity}

% general letter text fields
\setkomavar{subject}{\MyTitle{} \IfNonEmpty{\RecpJobRef}{\\ Referenz: \RecpJobRef}}
\setkomavar{signature}{\MySignature}
\setkomavar{place}{\MyCity}
\setkomavar{date}{\PrettyDate}

\begin{document}

\renewcommand*{\raggedsignature}{\raggedright} % shifts the signature block and greeting to the left

\begin{letter}{%
	\RecpCompany\\
	\IfNonEmpty{\RecpName}{ \RecpTitle \ \RecpName \\}
	\RecpStreet\\
	\RecpCity\\
	\RecpEmail
}
	\opening{\RecpSalutation}
	\medskip

	\RecpCoverLetter
	\bigskip

	\closing{Mit freundlichen Grüßen}
\end{letter}

\end{document}
